REPORT ZCURSO_009.

SELECTION-SCREEN BEGIN OF BLOCK bl.
  PARAMETERS p_cat TYPE c LENGTH 1.
SELECTION-SCREEN END OF BLOCK bl.

START-OF-SELECTION.

  DATA lv_categoria TYPE string.

  CASE p_cat.
    WHEN 'A'.
      lv_categoria = text-001.
    WHEN 'B'.
      lv_categoria = text-002.
    WHEN 'C'.
      lv_categoria = text-003.
    WHEN 'D'.
      lv_categoria = text-004.
    WHEN 'E'.
      lv_categoria = text-005.
    WHEN OTHERS.
      lv_categoria = text-006.
  ENDCASE.

  WRITE: / |Categoria selecionada: { lv_categoria }|.
