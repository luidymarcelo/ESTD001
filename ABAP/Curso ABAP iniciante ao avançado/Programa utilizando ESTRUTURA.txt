    , id TYPE int4
    , nome(20) TYPE C
    , email(50) TYPE C
    , END OF ls_cliente.

  DATA ld_nome(20)  TYPE  C.
  DATA ld_email(50) TYPE  C.

  ls_cliente-id = 1.
  ls_cliente-nome = 'João'.
  ls_cliente-email = 'joaos@tt.com'.

  ld_id   = 1.
  ld_nome	= 'João'.
  ld_email  = 'joao@jj.com'.

  WRITE ls_cliente-id.
